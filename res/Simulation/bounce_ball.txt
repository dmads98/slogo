//this part done
//set up
to setup
  clearscreen
  tell [1 2 3 4 5]                      ;; create 5 turtles
  ask [1 2 3 4 5] [ randomize ]         ;; place them randomly
  //reset-ticks                         ;; we could wait for this
end

//this part done
to randomize
  setxy random 400 random 400   ; 400 is the screen size
end

//this part done - main function
to go
  ask [1,2,3,4,5][
    pendown
    bounce
    forward 0.1
  ]
  //tick                                ;; we can create this latear, for now, we can make a go button that inputs go method to the console every second
end

// this part done
to bounce

  if equal xcor 400
    [ setheading minus heading) ]

  if equal xcor 0
    [ setheading minus heading) ]

  if equal ycor 400
    [ setheading difference 180 heading) ]

  if equal ycor 0
      [ setheading difference 180 heading) ]
end
